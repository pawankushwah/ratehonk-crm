    <script>
        const tenantId = 12;
        const testResults = {};
        
        function log(message, type = 'info') {
            const logs = document.getElementById('testLogs');
            const timestamp = new Date().toLocaleTimeString();
            const logEntry = `<div class="log-entry"><strong>[${timestamp}]</strong> ${message}</div>`;
            logs.innerHTML += logEntry;
            logs.scrollTop = logs.scrollHeight;
            console.log(`[${timestamp}] ${message}`);
        }

        function updateStatus(testName, status) {
            const indicator = document.getElementById(`status-${testName}`);
            if (indicator) {
                indicator.className = `status-indicator status-${status}`;
            }
            testResults[testName] = status;
            updateSummary();
        }

        function updateSummary() {
            const summary = document.getElementById('summaryResults');
            const total = Object.keys(testResults).length;
            const passed = Object.values(testResults).filter(s => s === 'success').length;
            const failed = Object.values(testResults).filter(s => s === 'error').length;
            const pending = total - passed - failed;
            
            summary.innerHTML = `
                <div class="grid">
                    <div><strong>Total Tests:</strong> ${total}</div>
                    <div style="color: #28a745;"><strong>Passed:</strong> ${passed}</div>
                    <div style="color: #dc3545;"><strong>Failed:</strong> ${failed}</div>
                    <div style="color: #ffc107;"><strong>Pending:</strong> ${pending}</div>
                </div>
            `;
        }

        async function configureFacebookCredentials() {
            try {
                log('🔧 Configuring Facebook Business Suite credentials...', 'info');
                
                const configData = {
                    appId: '1449283782943493',
                    appSecret: '4151640f8e36527dc32037aa1a7d868a',
                    enabled: true
                };

                const response = await fetch(`/api/tenants/${tenantId}/facebook/configure`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': 'Bearer test-token'
                    },
                    body: JSON.stringify(configData)
                });

                const result = await response.json();
                
                if (response.ok) {
                    log('✅ Facebook credentials configured successfully!', 'success');
                    document.getElementById('configResults').innerHTML = 
                        `<div class="success">✅ Configuration Success!<br>App ID: ${configData.appId}<br>Status: Enabled</div>`;
                    return true;
                } else {
                    log(`❌ Configuration failed: ${result.message || result.error}`, 'error');
                    document.getElementById('configResults').innerHTML = 
                        `<div class="error">❌ Configuration Failed<br>Error: ${result.message || result.error}</div>`;
                    return false;
                }
            } catch (error) {
                log(`❌ Configuration error: ${error.message}`, 'error');
                document.getElementById('configResults').innerHTML = 
                    `<div class="error">❌ Network Error: ${error.message}</div>`;
                return false;
            }
        }

        async function testEndpoint(endpoint) {
            try {
                log(`🔄 Testing ${endpoint} endpoint...`, 'info');
                updateStatus(endpoint, 'pending');
                
                const response = await fetch(`/api/tenants/${tenantId}/facebook/${endpoint}`, {
                    method: 'GET',
                    headers: {
                        'Authorization': 'Bearer test-token'
                    }
                });

                const result = await response.json();
                const resultDiv = document.getElementById(`result-${endpoint}`);
                
                if (response.ok) {
                    log(`✅ ${endpoint} endpoint working successfully!`, 'success');
                    updateStatus(endpoint, 'success');
                    resultDiv.innerHTML = `
                        <div class="success">
                            ✅ Success! 
                            <details>
                                <summary>View Response</summary>
                                <pre>${JSON.stringify(result, null, 2)}</pre>
                            </details>
                        </div>
                    `;
                } else {
                    log(`❌ ${endpoint} endpoint failed: ${result.message}`, 'error');
                    updateStatus(endpoint, 'error');
                    resultDiv.innerHTML = `
                        <div class="error">
                            ❌ Error: ${result.message || result.error}
                            <details>
                                <summary>View Response</summary>
                                <pre>${JSON.stringify(result, null, 2)}</pre>
                            </details>
                        </div>
                    `;
                }
            } catch (error) {
                log(`❌ ${endpoint} test error: ${error.message}`, 'error');
                updateStatus(endpoint, 'error');
                document.getElementById(`result-${endpoint}`).innerHTML = 
                    `<div class="error">❌ Network Error: ${error.message}</div>`;
            }
        }

        async function testOAuthFlow() {
            try {
                log('🔄 Testing Facebook OAuth flow...', 'info');
                updateStatus('oauth', 'pending');
                
                // Test OAuth URL generation
                const response = await fetch(`/api/tenants/${tenantId}/facebook/oauth`, {
                    method: 'GET',
                    headers: {
                        'Authorization': 'Bearer test-token'
                    }
                });

                const result = await response.json();
                
                if (response.ok && result.authUrl) {
                    log('✅ OAuth URL generated successfully!', 'success');
                    updateStatus('oauth', 'success');
                    document.getElementById('result-oauth').innerHTML = `
                        <div class="success">
                            ✅ OAuth URL Generated!
                            <br><a href="${result.authUrl}" target="_blank" class="btn-primary" style="margin-top: 10px;">Open OAuth Flow</a>
                            <details>
                                <summary>View OAuth URL</summary>
                                <pre>${result.authUrl}</pre>
                            </details>
                        </div>
                    `;
                } else {
                    log(`❌ OAuth test failed: ${result.message}`, 'error');
                    updateStatus('oauth', 'error');
                    document.getElementById('result-oauth').innerHTML = 
                        `<div class="error">❌ OAuth Error: ${result.message || result.error}</div>`;
                }
            } catch (error) {
                log(`❌ OAuth test error: ${error.message}`, 'error');
                updateStatus('oauth', 'error');
                document.getElementById('result-oauth').innerHTML = 
                    `<div class="error">❌ Network Error: ${error.message}</div>`;
            }
        }

        async function runAllTests() {
            log('🚀 Starting complete Facebook Business Suite test...', 'info');
            
            // Step 1: Configure credentials
            const configured = await configureFacebookCredentials();
            if (!configured) {
                log('❌ Cannot proceed without valid configuration', 'error');
                return;
            }
            
            // Wait a moment for configuration to settle
            await new Promise(resolve => setTimeout(resolve, 1000));
            
            // Step 2: Test all endpoints
            const endpoints = ['status', 'pages', 'insights', 'leads', 'sync'];
            for (const endpoint of endpoints) {
                await testEndpoint(endpoint);
                await new Promise(resolve => setTimeout(resolve, 500)); // Small delay between tests
            }
            
            // Step 3: Test advanced features
            await testOAuthFlow();
            await new Promise(resolve => setTimeout(resolve, 500));
            await testUnifiedSocial();
            
            log('🎉 Complete test suite finished!', 'success');
        }
